
#include <stdio.h>
#include <math.h>

int main(int argc, const char *argv[])
{
	// compute the number of digit of n in base 2
	int n = 0b1111101000;
	int count = 0;
	count= log2(n)+1;
	printf("Number of digit of %d (10) in base 2 is %d\n", n, count);

	// compute the number of digit of n in base 8
	int base = 8;
	count = log10(n) / log10(base) + 1;
	printf("Number of digit of %o(8) in base 8 is %d\n", n, count);

	// compute the number of digit of n in base 16
	base = 16;
	count = log10(n) / log10(base) + 1;
	printf("Number of digit of %X(16) in base %d is %d\n", n, base, count);

	// check that tan(alpha) = sin(alpha) / cos(alpha)
	float alpha = 60.0;
	alpha = alpha * M_PI / 180.0; // double : 1.0471975512, float 1.0471975803
	printf("alpha = %.10f\n", alpha);
	float tan_alpha = tan(alpha);
	float sin_alpha = sin(alpha);
	float cos_alpha = cos(alpha);
	printf("tan(%f) = %f\n", alpha, tan_alpha);
	printf("sin(%f) = %f\n", alpha, sin_alpha);
	printf("cos(%f) = %f\n", alpha, cos_alpha);
	printf("tan(%f) = sin(%f) / cos(%f) = %f\n", alpha, alpha, alpha, sin_alpha / cos_alpha);
	// display the difference between the two results
	printf("Difference: %e\n", tan_alpha - sin_alpha / cos_alpha);
	return 0;
}
